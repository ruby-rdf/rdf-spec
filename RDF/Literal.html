def inspect
  klass = self.class.to_s.match(/([^:]*)$/).to_s
  dt = self.datatype
  "\"#{escape(value)}\"#{('@' + self.language.to_s) if self.language?} R:L:(#{klass == 'Literal' && dt ? dt : klass})"
end
